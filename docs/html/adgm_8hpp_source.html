106  void movemesh()
107  {
108  // cycle over interior points
109  Walkdata dat;
110  cout << "DGmove: movemesh(): Calculating containing elements and area coordinates for each interior point\n";
111  for(int ipoin = 0; ipoin < ninpoin; ipoin++)
112  {
113  //cout << ipoin << endl;
114  // first find containing DG element by "walking-through" the DG
115  //cout << "DGmove: movemesh(): Point " << ipoin << endl;
116  dat = dg.find_containing_triangle_and_area_coords(points(ipoin,0), points(ipoin,1), dg.elems.size()/2);
117 
118  // store DG element and area coords in points
119  points(ipoin,2) = dat.elem;
120  for(int i = 0; i < ndim+1; i++)
121  points(ipoin,i+3) = dat.areacoords[i];
122  }
123 
124  // update coordinates in dgpoints using bmotion
125  cout << "DGmove: movemesh(): Moving the Delaunay graph\n";
126  int k = 0;
127  for(int i = 0; i < bmotion->rows(); i++)
128  {
129  for(int j = 0; j < ndim; j++)
130  dgpoints(i,j) += bmotion->get(i,j);
131  }
132 
133  // calculate new positions of interior points by mapping them to deformed DG elements using area coordinates calculated before
134  cout << "DGmove: movemesh(): Moving the interior points\n";
135  int elem;
136  for(int ipoin = 0; ipoin < ninpoin; ipoin++)
137  {
138  elem = points(ipoin,2);
139  for(int dim = 0; dim < ndim; dim++)
140  {
141  points(ipoin,dim) = 0.0;
142  for(int i = 0; i < ndim+1; i++)
143  points(ipoin,dim) += points(ipoin,i+3)*dgpoints(dginpoel(elem,i),dim);
144  }
145  }
