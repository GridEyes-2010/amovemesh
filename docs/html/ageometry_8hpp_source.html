232  void compute_splines()
233  {
234  // get point tangents first
235  std::cout << "HermiteSpline2D: compute_splines(): Getting tangents of boundary points." << std::endl;
236  for(int ipoin = 0; ipoin < m->gnbpoin(); ipoin++)
237  {
238  // in 2D, bpoints is not an array of stl std::vectors - we know there are only two faces surrounding each boundary point
239  // if edge is bounded by (x1,y1) and (x2,y2), tangent to edge is (x2-x1)i + (y2-y1)j
240 
241  // first check if this point belongs to a face that needs to be reconstructed
242  if(toRec(m->gbpoints(ipoin,1))==1 || toRec(m->gbpoints(ipoin,2))==1)
243  {
244 
245  std::vector<int> en(2);
246 
247  // get intfac faces containing this point
248  en[0] = m->gbpoints(ipoin,1);
249  en[1] = m->gbpoints(ipoin,2);
250  //std::cout << en[0] << " " << en[1] << std::endl;
251 
252  double xt, yt, dotx = 1, doty = 1, mag;
253 
254  // iterate over the two faces containing ipoin
255  for(int i = 0; i < 2; i++)
256  {
257  xt = m->gcoords(m->gintfac(en[i],3),0) - m->gcoords(m->gintfac(en[i],2),0);
258  yt = m->gcoords(m->gintfac(en[i],3),1) - m->gcoords(m->gintfac(en[i],2),1);
259  mag = sqrt(xt*xt + yt*yt);
260  ptangents(m->gbpoints(ipoin,0),0) += xt;
261  ptangents(m->gbpoints(ipoin,0),1) += yt;
262  dotx *= xt/mag;
263  doty *= yt/mag;
264  }
265  ptangents(m->gbpoints(ipoin,0),0) /= 2.0;
266  ptangents(m->gbpoints(ipoin,0),1) /= 2.0;
267 
268  // now normalize the averaged tangent std::vector
269  mag = sqrt(ptangents(m->gbpoints(ipoin,0),0)*ptangents(m->gbpoints(ipoin,0),0) + ptangents(m->gbpoints(ipoin,0),1)*ptangents(m->gbpoints(ipoin,0),1));
270  ptangents(m->gbpoints(ipoin,0),0) /= mag;
271  ptangents(m->gbpoints(ipoin,0),1) /= mag;
272 
273  // check if this is a corner point
274  double dotp = dotx+doty; // dot product of tangents of the two faces
275  if(dotp < cos(angle_threshold)) {
276  isCorner(m->gbpoints(ipoin,0)) = 1; // set isCorner for global point number corresponding to boundary point ipoin
277  std::cout << "Boundary point " << m->gbpoints(ipoin,0) << " is a corner point!" << std::endl;
278  }
279  }
280  }
281 
282  // iterate over boundary faces of the mesh
283  std::cout << "HermiteSpline2D: compute_splines(): Iterating over boundary faces to get tangents." << std::endl;
284  std::vector<int> pnts(m->gnnofa()); // to store point numbers of points making up iface
285  for(int iface = 0; iface < m->gnbface(); iface++)
286  {
287  if(toRec(iface) == 1)
288  {
289  // get tangent for this face
290  for(int inofa = 0; inofa < m->gnnofa(); inofa++)
291  pnts[inofa] = m->gintfac(iface,inofa+2);
292 
293  // magnitude of tangent (x2-x1)i + (y2-y1)j
294  double mag = sqrt((m->gcoords(pnts[1],0)-m->gcoords(pnts[0],0))*(m->gcoords(pnts[1],0)-m->gcoords(pnts[0],0)) + (m->gcoords(pnts[1],1)-m->gcoords(pnts[0],1))*(m->gcoords(pnts[1],1)-m->gcoords(pnts[0],1)));
295 
296  for(int inofa = 0; inofa < m->gnnofa(); inofa++)
297  {
298  if(isCorner(pnts[inofa]) == 0) // if not a corner
299  {
300  gallfa(iface,2*inofa) = ptangents(pnts[inofa],0);
301  gallfa(iface,2*inofa+1) = ptangents(pnts[inofa],1);
302  }
303  else // if pnts[inofa] is a corner, use this face's tangent at that point
304  {
305  gallfa(iface,2*inofa) = (m->gcoords(pnts[1],0) - m->gcoords(pnts[0],0))/mag;
306  gallfa(iface,2*inofa+1) = (m->gcoords(pnts[1],1) - m->gcoords(pnts[0],1))/mag;
307  }
308  }
309 
310  /* Note that gallfa(*,0) contains x-component of tangent at point 1, gallfa(*,1) contains y-component of tangent at point 1,
311  gallfa(*,2) contains x-component of tangent at point 2, and gallfa(*,3) contains y-component of tangent at points 2. */
312  }
313  }
