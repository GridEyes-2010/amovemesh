200 void CSpline::sequence()
201 {
202  if(!issequenced)
203  {
204  int curpoin = 0, nextpoin = 0, startpoin, startface, curface, nextface;
205 
206  // first find a face that belongs to the boundary to be reconstructed
207  for(int iface = 0; iface < m->gnface(); iface++)
208  if(toRec(iface) == 1) startface = iface;
209 
210  int nf = 0, np = 0; // number of faces and number of points respectively
211 
212  if(isClosed)
213  {
214  nextface = startface;
215  while(true)
216  {
217  curface = nextface;
218  seq_bface(curface) = nf;
219  segface(nf) = curface;
220  seq_spoin(np) = m->gbface(curface,0);
221  nf++; np++;
222 
223  nextpoin = m->gbface(curface,1);
224 
225  // search for the other face which contains nextpoin
226  nextface = -1;
227  for(int iface = 0; iface < m->gnface(); iface++)
228  if(toRec(iface) == 1 && m->gbface(iface,0) == nextpoin) nextface = iface;
229  if(nextface == -1) std::cout << "UMesh2d: sequence(): ! Point not found!" << std::endl;
230  if(nextface == startface) break;
231  }
232  seq_spoin(np) = seq_spoin(0); // close the point list
233  }
234  else // curve is open
235  {
236  nextface = startface;
237 
238  //first find the last element
239  std::cout << "UMesh2d: sequence(): Finding first face of open boundary." << std::endl;
240  while(true)
241  {
242  curface = nextface;
243  nextpoin = m->gbface(curface,0);
244 
245  // search for the other face which contains nextpoin
246  nextface = -1;
247  for(int iface = 0; iface < m->gnface(); iface++)
248  if(toRec(iface) == 1 && m->gbface(iface,1) == nextpoin) nextface = iface;
249 
250  if(nextface == -1) {
251  std::cout << "UMesh2d: sequence(): Reached start of boundary to be reconstructed." << std::endl;
252  break;
253  }
254  }
255 
256  nextface = curface;
257 
258  // now that the first face has been found, sequence the boundary
259  while(true)
260  {
261  curface = nextface;
262  seq_bface(curface) = nf;
263  segface(nf) = curface;
264  seq_spoin(np) = m->gbface(curface,0);
265  nf++; np++;
266 
267  nextpoin = m->gbface(curface,1);
268 
269  // search for the other face which contains nextpoin
270  nextface = -1;
271  for(int iface = 0; iface < m->gnface(); iface++)
272  if(toRec(iface) == 1 && m->gbface(iface,0) == nextpoin) nextface = iface;
273 
274  if(nextface == -1) {
275  std::cout << "UMesh2d: sequence(): " << nf << " boundary faces and " << np << " boundary points sequenced." << std::endl;
276  if(nf != nseg) std::cout << "UMesh2d: sequence(): ! All faces are not accounted for!" << std::endl;
277  break;
278  }
279  }
280 
281  // finally, add last point
282  seq_spoin(np) = nextpoin;
283  }
284  }
285 
286  else
287  {
288  if(!face_list_available) {
289  std::cout << "CSpline: sequence(): Boundary part is sequenced, but facelist is not available!!" << std::endl;
290  return;
291  }
292 
293  // facelist is already sequenced; just use it.
294  for(int iseg = 0; iseg < nseg; iseg++)
295  {
296  seq_bface(facelist[iseg]) = iseg;
297  segface(iseg) = facelist[iseg];
298  seq_spoin(iseg) = m->gbface(facelist[iseg],0);
299  }
300  seq_spoin(nseg) = m->gbface(facelist[nseg-1], 1);
301  }

304 void CSpline::compute()
305 {
306  if(isClosed)
307  {
308  // Assemble RHS std::vector for each dimension
309  for(int idim = 0; idim < dim; idim++)
310  {
311  srhs[idim](0) = 3.0*(m->gcoords(seq_spoin(1),idim) - m->gcoords(seq_spoin(nseg-1),idim));
312  for(int i = 1; i < nseg; i++)
313  {
314  srhs[idim](i) = 3.0*(m->gcoords(seq_spoin(i+1),idim) - m->gcoords(seq_spoin(i-1),idim));
315  }
316  srhs[idim](nseg) = 3.0*(m->gcoords(seq_spoin(1),idim) - m->gcoords(seq_spoin(nseg-1),idim));
317  }
318 
319  // Assemble LHS matrix
320  slhs.set(0,0, 4.0);
321  slhs.set(0,1, 1.0);
322  slhs.set(0,nseg-1, 1.0);
323 
324  for(int i = 1; i < nseg; i++)
325  {
326  slhs.set(i,i-1, 1.0);
327  slhs.set(i,i, 4.0);
328  slhs.set(i,i+1, 1.0);
329  }
330 
331  slhs.set(nseg,1, 1.0);
332  slhs.set(nseg,nseg-1, 1.0);
333  slhs.set(nseg,nseg, 4.0);
334  }
335 
336  else
337  {
338  // Assemble RHS std::vector for each dimension
339  for(int idim = 0; idim < dim; idim++)
340  {
341  srhs[idim](0) = 3.0*(m->gcoords(seq_spoin(1),idim) - m->gcoords(seq_spoin(0),idim));
342  for(int i = 1; i < nseg; i++)
343  {
344  srhs[idim](i) = 3.0*(m->gcoords(seq_spoin(i+1),idim) - m->gcoords(seq_spoin(i-1),idim));
345  }
346  srhs[idim](nseg) = 3.0*(m->gcoords(seq_spoin(nseg),idim) - m->gcoords(seq_spoin(nseg-1),idim));
347  }
348 
349  // Assemble LHS matrix
350  slhs.set(0,0, 2.0);
351  slhs.set(0,1, 1.0);
352 
353  for(int i = 1; i < nseg; i++)
354  {
355  slhs.set(i,i-1, 1.0);
356  slhs.set(i,i, 4.0);
357  slhs.set(i,i+1, 1.0);
358  }
359 
360  slhs.set(nseg,nseg-1, 1.0);
361  slhs.set(nseg,nseg, 2.0);
362  }
363 
364  // solve
365  std::cout << "CSpline: compute(): Solving linear systems" << std::endl;
366  amat::Matrix<double> d0(nseg+1,1);
367  d0.zeros();
368  for(int idim = 0; idim < dim; idim++)
369  D[idim] = sparseCG_d(&slhs, srhs[idim], d0, tol, maxiter);
370 
371  // get coeffs
372  //std::cout << "CSpline: compute(): Getting spline coefficients" << std::endl;
373  double yi, yip;
374  for(int idim = 0; idim < dim; idim++)
375  {
376  for(int i = 0; i < nseg; i++)
377  {
378  // first get idim-coordinates of the ith and (i+1)th spline points
379  yi = m->gcoords(seq_spoin.get(i),idim);
380  yip = m->gcoords(seq_spoin.get(i+1),idim);
381 
382  scf[idim](i,0) = yi;
383  scf[idim](i,1) = D[idim](i);
384  scf[idim](i,2) = 3*(yip - yi) - 2*D[idim](i) - D[idim].get(i+1);
385  scf[idim](i,3) = 2*(yi - yip) + D[idim](i) + D[idim].get(i+1);
386  }
387  }
388  std::cout << "CSpline: compute(): Done." << std::endl;
389 }

552 void BoundaryReconstruction2d::detect_corners()
553 {
554  int p1, p2, p3;
555  double n1x, n1y, n2x, n2y, mag1, mag2, dp;
556 
557  std::cout << "BoundaryReconstruction2d: detect_corners(): Searching each part for corners" << std::endl;
558 
559  // Scan each part. At each boundary point, calculate dot product of the two normals of the two faces containing that point.
560  for(int ipart = 0; ipart < nparts; ipart++)
561  {
562  for(int ipoin = 0; ipoin < m->gnbpoin(); ipoin++)
563  {
564  // check left and right faces
565  if(toRec(ipart,m->gbpointsb(ipoin,1)) && toRec(ipart,m->gbpointsb(ipoin,2)))
566  {
567  p1 = m->gbface(m->gbpointsb(ipoin,1),0);
568  p2 = m->gbpointsb(ipoin,0);
569  p3 = m->gbface(m->gbpointsb(ipoin,2),1);
570 
571  // check dot-product of normals
572  n1x = m->gcoords(p2,1) - m->gcoords(p1,1);
573  n1y = m->gcoords(p1,0) - m->gcoords(p2,0);
574  n2x = m->gcoords(p3,1) - m->gcoords(p2,1);
575  n2y = m->gcoords(p2,0) - m->gcoords(p3,0);
576 
577  mag1 = sqrt(n1x*n1x + n1y*n1y);
578  mag2 = sqrt(n2x*n2x + n2y*n2y);
579  n1x /= mag1;
580  n1y /= mag1;
581  n2x /= mag2;
582  n2y /= mag2;
583 
584  dp = n1x*n2x + n1y*n2y;
585 
586  if(dp < cos(cangle))
587  {
588  std::vector<int> c(4);
589  c[0] = p2; // global point number
590  c[1] = ipoin; // boundary point number
591  c[2] = m->gbpointsb(ipoin,1);
592  c[3] = m->gbpointsb(ipoin,2);
593  corners[ipart].push_back(c);
594  ncorners[ipart]++;
595  }
596  }
597  }
598  std::cout << "BoundaryReconstruction2d: detect_corners(): Part " << ipart << " contains " << ncorners[ipart] << " corner(s)" << std::endl;
599  std::cout << " at points ";
600  for(int i = 0; i < ncorners[ipart]; i++)
601  std::cout << corners[ipart][i][0] << ", ";
602  std::cout << std::endl;
603  }
604 }

642 void BoundaryReconstruction2d::split_parts()
643 {
644  nnparts = 0; // update to get total number of parts after all splits
645  std::cout << "BoundaryReconstruction2d: split_parts(): Splitting parts" << std::endl;
646 
647  // Split parts containing corners
648  for(int ipart = 0; ipart < nparts; ipart++)
649  {
650  std::vector<int> facelist;
651 
652  if(ncorners[ipart] == 0)
653  {
654  int nextface = startface[ipart], curface, nextpoint;
655 
656  // traverse through faces in this part
657  while(true)
658  {
659  curface = nextface;
660  facelist.push_back(curface);
661  facepart(curface,0) = nnparts;
662  facepart(curface,1) = facelist.size()-1;
663 
664  nextpoint = m->gbfacebp(curface,1); // get boundary point number of next point (NOT global point number)
665  nextface = m->gbpointsb(nextpoint,2);
666 
667  if(toRec(ipart,nextface) == 0 || nextface == startface[ipart])
668  { break;
669  }
670  }
671 
672  partfaces.push_back(facelist);
673  isSplitClosed.push_back(isClosed[ipart]);
674  nnparts++;
675  continue;
676  }
677 
678  if(isClosed[ipart])

909  void compute_splines()
910  {
911  // get point tangents first
912  std::cout << "HermiteSpline2D: compute_splines(): Getting tangents of boundary points." << std::endl;
913  for(int ipoin = 0; ipoin < m->gnbpoin(); ipoin++)
914  {
915  // in 2D, bpoints is not an array of stl std::vectors - we know there are only two faces surrounding each boundary point
916  // if edge is bounded by (x1,y1) and (x2,y2), tangent to edge is (x2-x1)i + (y2-y1)j
917 
918  // first check if this point belongs to a face that needs to be reconstructed
919  if(toRec(m->gbpoints(ipoin,1))==1 || toRec(m->gbpoints(ipoin,2))==1)
920  {
921 
922  std::vector<int> en(2);
923 
924  // get intfac faces containing this point
925  en[0] = m->gbpoints(ipoin,1);
926  en[1] = m->gbpoints(ipoin,2);
927  //std::cout << en[0] << " " << en[1] << std::endl;
928 
929  double xt, yt, dotx = 1, doty = 1, mag;
930 
931  // iterate over the two faces containing ipoin
932  for(int i = 0; i < 2; i++)
933  {
934  xt = m->gcoords(m->gintfac(en[i],3),0) - m->gcoords(m->gintfac(en[i],2),0);
935  yt = m->gcoords(m->gintfac(en[i],3),1) - m->gcoords(m->gintfac(en[i],2),1);
936  mag = sqrt(xt*xt + yt*yt);
937  ptangents(m->gbpoints(ipoin,0),0) += xt;
938  ptangents(m->gbpoints(ipoin,0),1) += yt;
939  dotx *= xt/mag;
940  doty *= yt/mag;
941  }
942  ptangents(m->gbpoints(ipoin,0),0) /= 2.0;
943  ptangents(m->gbpoints(ipoin,0),1) /= 2.0;
944 
945  // now normalize the averaged tangent std::vector
946  mag = sqrt(ptangents(m->gbpoints(ipoin,0),0)*ptangents(m->gbpoints(ipoin,0),0) + ptangents(m->gbpoints(ipoin,0),1)*ptangents(m->gbpoints(ipoin,0),1));
947  ptangents(m->gbpoints(ipoin,0),0) /= mag;
948  ptangents(m->gbpoints(ipoin,0),1) /= mag;
949 
950  // check if this is a corner point
951  double dotp = dotx+doty; // dot product of tangents of the two faces
952  if(dotp < cos(angle_threshold)) {
953  isCorner(m->gbpoints(ipoin,0)) = 1; // set isCorner for global point number corresponding to boundary point ipoin
954  std::cout << "Boundary point " << m->gbpoints(ipoin,0) << " is a corner point!" << std::endl;
955  }
956  }
957  }
958 
959  // iterate over boundary faces of the mesh
960  std::cout << "HermiteSpline2D: compute_splines(): Iterating over boundary faces to get tangents." << std::endl;
961  std::vector<int> pnts(m->gnnofa()); // to store point numbers of points making up iface
962  for(int iface = 0; iface < m->gnbface(); iface++)
963  {
964  if(toRec(iface) == 1)
965  {
966  // get tangent for this face
967  for(int inofa = 0; inofa < m->gnnofa(); inofa++)
968  pnts[inofa] = m->gintfac(iface,inofa+2);
969 
970  // magnitude of tangent (x2-x1)i + (y2-y1)j
971  double mag = sqrt((m->gcoords(pnts[1],0)-m->gcoords(pnts[0],0))*(m->gcoords(pnts[1],0)-m->gcoords(pnts[0],0)) + (m->gcoords(pnts[1],1)-m->gcoords(pnts[0],1))*(m->gcoords(pnts[1],1)-m->gcoords(pnts[0],1)));
972 
973  for(int inofa = 0; inofa < m->gnnofa(); inofa++)
974  {
975  if(isCorner(pnts[inofa]) == 0) // if not a corner
976  {
977  gallfa(iface,2*inofa) = ptangents(pnts[inofa],0);
978  gallfa(iface,2*inofa+1) = ptangents(pnts[inofa],1);
979  }
980  else // if pnts[inofa] is a corner, use this face's tangent at that point
981  {
982  gallfa(iface,2*inofa) = (m->gcoords(pnts[1],0) - m->gcoords(pnts[0],0))/mag;
983  gallfa(iface,2*inofa+1) = (m->gcoords(pnts[1],1) - m->gcoords(pnts[0],1))/mag;
984  }
985  }
986 
987  /* Note that gallfa(*,0) contains x-component of tangent at point 1, gallfa(*,1) contains y-component of tangent at point 1,
988  gallfa(*,2) contains x-component of tangent at point 2, and gallfa(*,3) contains y-component of tangent at points 2. */
989  }
990  }
991 
992  // iterate over boundary faces of mesh again to calculate geometric spline coeffs from gallfa and intfac
993  std::cout << "HermiteSpline2d: compute_splines(): Iterating over boundary faces again to compute geometric coefficients of the splines" << std::endl;
994  for(int iface = 0; iface < m->gnbface(); iface++)
995  if(toRec(iface) == 1)
996  {
997  for(int idim = 0; idim < m->gndim(); idim++)
998  {
999  cf[idim](iface,0) = m->gcoords(m->gintfac(iface,2),idim);
1000  cf[idim](iface,1) = m->gcoords(m->gintfac(iface,3),idim);
1001  cf[idim](iface,2) = gallfa(iface,idim);
1002  cf[idim](iface,3) = gallfa(iface,idim+2);
1003  }
1004  }
1005  }
